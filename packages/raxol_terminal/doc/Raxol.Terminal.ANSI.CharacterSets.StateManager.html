    charset_state()

    @type charset_state() :: %{
  active: charset(),
  single_shift: charset() | nil,
  g0: charset(),
  g1: charset(),
  g2: charset(),
  g3: charset(),
  gl: :g0 | :g1 | :g2 | :g3,
  gr: :g0 | :g1 | :g2 | :g3
}
